function product_rule {
    param($a, $b)
    return $a**$b
}

function quotient_rule {
    param($a, $b)
    return $a**$b
}

function power_rule {
    param($a, $b)
    return $a**$b
}

function zero_rule {
    param($a)
    return $a**0
}

function negative_rule {
    param($a, $b)
    return 1/($a**$b)
}

$questions = @(
    ("1. What is 5^3 ร 5^4? ", product_rule(5, 3 + 4)),
    ("2. What is 10^5 รท 10^2? ", quotient_rule(10, 5 - 2)),
    ("3. What is (3^2)^4? ", power_rule((3**2), 4)),
    ("4. What is 7^0? ", zero_rule(7)),
    ("5. What is 2^-3? ", negative_rule(2, 3))
)

$score = 0

foreach ($question in $questions) {
    $user_answer = Read-Host $question[0]
    if ($user_answer -eq $question[1]) {
        Write-Host "Correct!"
        $score++
    } else {
        Write-Host ("Incorrect. The correct answer is " + $question[1])
    }
}

Write-Host ("You got " + $score + "/5 questions correct. Well done!")
